29def init_argparser(): 

30 # Set up an argument parser. We are using defaults to stay consistent with other software. 

31 # description gets added to the usage statements 

32 argument_parser = argparse.ArgumentParser( 

33 formatter_class=argparse.RawDescriptionHelpFormatter, 

34 description=DESCRIPTION, 

35 prog="irida-uploader -d DIRECTORY", 

36 epilog='-c* options can be used without a parameter to prompt for input.') 

37 # Our main argument. It is required or else an error will be thrown when the program is run 

38 # Normally we would use a positional argument, but because of our 1 or None config overrides it makes sense to use 

39 # a optional argument, with the required set to True. We have to suppress the help on the argument, and add it's 

40 # help information to the formatted description text of the parser for the argument to be shown as required when 

41 # --help is used. 

42 argument_parser.add_argument('-d', '--directory', 

43 action='store', 

44 required=True, 

45 help=argparse.SUPPRESS) 

46 # help='Location of sequencing run to upload. Directory must be writable.') 

47 # Add the version argument 

48 argument_parser.add_argument('--version', 

49 action='version', version='IRIDA Uploader {}'.format(VERSION_NUMBER)) 

50 # Optional argument, for using an alternative config file. 

51 argument_parser.add_argument('-c', '--config', 

52 action='store', 

53 help='Path to an alternative configuration file. ' 

54 'This overrides the default config file in the config directory') 

55 # Optional argument, Force uploading a run even if a non new status file exists 

56 argument_parser.add_argument('-f', '--force', 

57 action='store_true', # This line makes it not parse a variable 

58 help='Uploader will ignore the status file, ' 

59 'and try to upload even when a run is in non new status.') 

60 # Optional argument, Force uploading a run even if a non new status file exists 

61 argument_parser.add_argument('-p', '--continue_partial', 

62 action='store_true', # This line makes it not parse a variable 

63 help='Uploader will continue an existing upload run when upload status is partial.') 

64 # Optional argument, Upload all sequencing runs in a directory of runs 

65 argument_parser.add_argument('-b', '--batch', 

66 action='store_true', # This line makes it not parse a variable 

67 help='Uploader will expect a directory containing a sequencing run directories, ' 

68 'and upload in batch. ' 

69 'The list of runs is generated at start time ' 

70 '(Runs added to directory mid upload will not be uploaded).') 

71 # Optional argument, Upload mode 

72 argument_parser.add_argument('-u', '--upload_mode', 

73 action='store', 

74 help='Choose which upload mode to use. ' 

75 'Supported modes: ' + str(UPLOAD_MODES)) 

76 

77 # Optional arguments for overriding config file settings 

78 # Explanation: 

79 # nargs='?', const=True, default=False, 

80 # Allows zero or one parameters 

81 # If the argument is not given: the value will be False (indicates load from file) 

82 # If the argument is given, and no parameter given: the value will be True (prompt user for input) 

83 # If the argument is given, and parameter is given: the value will be the parameter (used to override config) 

84 argument_parser.add_argument('-ci', '--config_client_id', action='store', nargs='?', const=True, default=False, 

85 help='Override the "client_id" field in config file. ' 

86 'This is for the uploader client created in IRIDA, ' 

87 'used to handle the data upload') 

88 argument_parser.add_argument('-cs', '--config_client_secret', action='store', nargs='?', const=True, default=False, 

89 help='Override "client_secret" in config file. ' 

90 'This is for the uploader client created in IRIDA, ' 

91 'used to handle the data upload') 

92 argument_parser.add_argument('-cu', '--config_username', action='store', nargs='?', const=True, default=False, 

93 help='Override "username" in config file. This is your IRIDA account username.') 

94 argument_parser.add_argument('-cp', '--config_password', action='store', nargs='?', const=True, default=False, 

95 help='Override "password" in config file. This corresponds to your IRIDA account.') 

96 argument_parser.add_argument('-cb', '--config_base_url', action='store', nargs='?', const=True, default=False, 

97 help='Override "base_url" in config file. The api url for your irida instance ' 

98 '(example: https://my.irida.server/api/)') 

99 argument_parser.add_argument('-cr', '--config_parser', action='store', nargs='?', const=True, default=False, 

100 help='Override "parser" in config file. ' 

101 'Choose the type of sequencer data that is being uploaded. ' 

102 'Supported parsers: ' + str(supported_parsers)) 

103 argument_parser.add_argument('-r', '--readonly', 

104 action='store_true', # This line makes it not parse a variable 

105 help='Upload in Read Only mode, does not write status or log file to run directory.') 

106 argument_parser.add_argument('-cd', '--delay', action='store', nargs='?', const=True, default=False, 

107 help='Accepts an Integer for minutes to delay. When set, new runs will have their ' 

108 'status set to delayed. When uploading a run with the delayed status, the run ' 

109 'will only upload if the given amount of minutes has passes since it was set to ' 

110 'delayed. Default = 0: When set to 0, runs will not be given delayed status.') 

111 argument_parser.add_argument('-ct', '--config_timeout', action='store', nargs='?', const=True, default=False, 

112 help='Accepts an Integer for the expected transfer time in seconds per MB. ' 

113 'Default is 10 second for every MB of data to transfer (100kb/s). Increasing this' 

114 ' number can reduce timeout errors when your transfer speed is very slow.') 

115 argument_parser.add_argument('-fs', '--minimum_file_size', action='store', nargs='?', const=True, default=False, 

116 help='Accepts an Integer for the minimum file size in KB. Default is 0 KB. Files that ' 

117 'are too small will appear as an error during run validation.') 

118 return argument_parser

121def _set_config_override(args): 

122 """ 

123 Check the config override arguments and override 

124 :param args: list of args from parseargs 

125 :return: 

126 """ 

127 client_id = None 

128 client_secret = None 

129 username = None 

130 password = None 

131 base_url = None 

132 parser = None 

133 readonly = None 

134 delay = None 

135 timeout = None 

136 minimum_file_size = None 

137 

138 if args.config_client_id is True: 

139 print("Enter Client ID:") 

140 client_id = input() 

141 elif args.config_client_id is not False: 

142 client_id = args.config_client_id 

143 

144 if args.config_client_secret is True: 

145 print("Enter Client Secret:") 

146 # getpass blanks out the secret entry 

147 client_secret = getpass.getpass() 

148 elif args.config_client_secret is not False: 

149 client_secret = args.config_client_secret 

150 

151 if args.config_username is True: 

152 print("Enter Username:") 

153 username = input() 

154 elif args.config_username is not False: 

155 username = args.config_username 

156 

157 if args.config_password is True: 

158 print("Enter Password:") 

159 # getpass blanks out password entry 

160 password = getpass.getpass() 

161 elif args.config_password is not False: 

162 password = args.config_password 

163 

164 if args.config_base_url is True: 

165 print("Enter Base IRIDA URL (format: http://my.irida.server/api/):") 

166 base_url = input() 

167 elif args.config_base_url is not False: 

168 base_url = args.config_base_url 

169 

170 if args.config_parser is True: 

171 print("Enter Parser to use:") 

172 parser = input() 

173 elif args.config_parser is not False: 

174 parser = args.config_parser 

175 

176 if args.readonly is True: 

177 readonly = args.readonly 

178 

179 if args.delay is True: 

180 print("Enter delay in minutes (Integer):") 

181 delay = int(input()) 

182 elif args.delay is not False: 

183 delay = int(args.delay) 

184 

185 if args.config_timeout is True: 

186 print("Enter timeout per MB in seconds (Integer):") 

187 timeout = int(input()) 

188 elif args.config_timeout is not False: 

189 timeout = int(args.config_timeout) 

190 

191 if args.minimum_file_size is True: 

192 print("Enter minimum file size in KB (Integer):") 

193 minimum_file_size = int(input()) 

194 elif args.minimum_file_size is not False: 

195 minimum_file_size = int(args.minimum_file_size) 

196 

197 config.set_config_options(client_id=client_id, 

198 client_secret=client_secret, 

199 username=username, 

200 password=password, 

201 base_url=base_url, 

202 parser=parser, 

203 readonly=readonly, 

204 delay=delay, 

205 timeout=timeout, 

206 minimum_file_size=minimum_file_size)

209def _config_uploader(args): 

210 """ 

211 Sets up config settings for command line uploading 

212 :param args: 

213 :return: 

214 """ 

215 # If a config file is passed in, set it before starting upload 

216 if args.config: 

217 config.set_config_file(args.config) 

218 # config.user_config_file = args.config 

219 # Init config 

220 config.setup() 

221 # Override with any passed in options 

222 _set_config_override(args)

225def main(): 

226 # Parse the arguments passed from the command line and start the upload 

227 argument_parser = init_argparser() 

228 args = argument_parser.parse_args() 

229 # Setup config options 

230 _config_uploader(args) 

231 

232 # Verify directory is readable before upload 

233 if not os.access(args.directory, os.R_OK): # Cannot access upload directory 

234 print("ERROR! Specified directory is not readable: {}".format(args.directory)) 

235 return 1 

236 

237 # Verify force and continue are not both active 

238 if args.force and args.continue_partial: 

239 print("ERROR! --force and --continue_partial are mutually exclusive. " 

240 "To continue a partial upload use --continue_partial. " 

241 "To upload all samples from the beginning use --force") 

242 return 1 

243 

244 # Start Upload 

245 if args.batch: 

246 return upload_batch(args.directory, args.force, args.upload_mode, args.continue_partial) 

247 else: 

248 return upload(args.directory, args.force, args.upload_mode, args.continue_partial)
