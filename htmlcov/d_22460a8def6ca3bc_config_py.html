29def _init_config_parser(): 

30 """ 

31 Creates the RawConfigParser object, adds the fields and fills with default/empty values 

32 :return: 

33 """ 

34 global _conf_parser 

35 # Create a new config parser 

36 _conf_parser = RawConfigParser() 

37 # Put all options under a Settings Header 

38 _conf_parser.add_section("Settings") 

39 

40 # Create defaults tuple 

41 SettingsDefault = namedtuple('SettingsDefault', ['setting', 'default_value']) 

42 default_settings = [SettingsDefault._make(["client_id", ""]), 

43 SettingsDefault._make(["client_secret", ""]), 

44 SettingsDefault._make(["username", ""]), 

45 SettingsDefault._make(["password", ""]), 

46 SettingsDefault._make(["base_url", ""]), 

47 SettingsDefault._make(["parser", "directory"]), 

48 SettingsDefault._make(["readonly", False]), 

49 SettingsDefault._make(["delay", 0]), 

50 SettingsDefault._make(["timeout", 10]), # default timeout scale is 10 seconds per mb 

51 SettingsDefault._make(["minimum_file_size", 0]), # default minimum file size in kb 

52 SettingsDefault._make(["http_max_retries", 5]), 

53 SettingsDefault._make(["http_backoff_factor", 0]), 

54 ] 

55 # add defaults to config parser 

56 for config in default_settings: 

57 _conf_parser.set("Settings", config.setting, config.default_value)

60def _create_new_config_file(): 

61 """ 

62 Tries to create a new config file 

63 Exit's if the file already exists, or a user config file is already set 

64 This only gets used to create a fresh default configuration file 

65 Fills the new file with defaults 

66 :return: 

67 """ 

68 global _user_config_file 

69 if _user_config_file is not None: 

70 logging.error("Trying to create new config file when config file as already been set. Exiting") 

71 return 

72 

73 _user_config_file = _default_user_config_file_path 

74 

75 # Create path to file if it doesn't exit yet 

76 if not os.path.exists(os.path.dirname(_user_config_file)): 

77 os.makedirs(os.path.dirname(_user_config_file)) 

78 # return if the file already exists 

79 if os.path.exists(_user_config_file): 

80 logging.error("File already exits, cannot create a new file") 

81 return 

82 

83 # init the config parser with default values 

84 global _conf_parser 

85 with open(_user_config_file, 'w') as config_file: 

86 _conf_parser.write(config_file) 

87 

88 logging.info("Config File Created: " + str(_user_config_file)) 

89 logging.info("Please edit your config file to connect to IRIDA")

236def read_config_option(key, expected_type=None, default_value=None): 

237 """Read the specified value from the configuration file. 

238 

239 Args: 

240 key: the name of the key to read from the config file. 

241 expected_type: read the config option as the specified type (if specified) 

242 default_value: if the key doesn't exist, just return the default value. 

243 If the default value is not specified, the function will throw whatever 

244 error was raised by the configuration parser 

245 """ 

246 logging.debug("Reading config option {} with expected type {}".format(key, expected_type)) 

247 

248 global _conf_parser

250 if not expected_type: 

251 value = _conf_parser.get("Settings", key)

254 elif expected_type is int:

256 logging.debug("Got configuration for key {}: {}".format(key, res)) 

257 # Return int, or string evaluated to int, or NameError exception otherwise

259 return res 

260 elif type(res) is str: 

261 try: 

262 return int(res) 

263 except Exception:

265 logging.error(error_msg) 

266 raise NameError(error_msg) 

267 elif expected_type is float: 

268 res = _conf_parser.get("Settings", key) 

269 logging.debug("Got configuration for key {}: {}".format(key, res)) 

270 # Return int, or string evaluated to int, or NameError exception otherwise 

271 if type(res) is float: 

272 return res 

273 elif type(res) is str: 

274 try: 

275 return float(res) 

276 except Exception: 

277 error_msg = f"Config file field '{key}' expected float but instead got '{res}'" 

278 logging.error(error_msg) 

279 raise NameError(error_msg) 

280 elif expected_type is bool: 

281 res = _conf_parser.get("Settings", key) 

282 logging.debug("Got configuration for key {}: {}".format(key, res)) 

283 # Return bool, or string evaluated to bool, or NameError exception otherwise

289 raise NameError

297def _eval_boolean(string, field): 

298 """ 

299 Converts string to bool value, accepts any case permutation. 

300 If non true/false is given, displays error to user and raises to exit 

301 :param string: string to evaluate as a boolean 

302 :param field: config field the key originated from for logging and error message 

303 :return: Boolean 

304 """ 

305 caps = string.upper() 

306 if caps == "TRUE": 

307 logging.debug("Evaluated as True") 

308 return True 

309 elif caps == "FALSE": 

310 logging.debug("Evaluated as False") 

311 return False 

312 else: 

313 error_msg = "Config file field '{}' expected 'True' or 'False' but instead got '{}'".format(field, string) 

314 logging.error(error_msg) 

315 raise NameError(error_msg)
